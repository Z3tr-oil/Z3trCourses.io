Python Functions

Reusable blocks of code in Python

Defining a Function


def greet():
    print("Hello, world!")

greet()

Function with Parameters


def add(a, b):
    return a + b

result = add(3, 5)
print(result)  # 8

Default Parameters


def greet(name="Guest"):
    print("Hello,", name)

greet("Alice")
greet()

Return Values


def square(x):
    return x * x

print(square(4))  # 16

Docstrings


def multiply(a, b):
    """Returns the product of a and b."""
    return a * b

print(multiply(2, 3))

Multiple Return Values


def min_max(numbers):
    return min(numbers), max(numbers)

mn, mx = min_max([1, 2, 3])
print(mn, mx)  # 1 3

Keyword Arguments


def greet(name, msg):
    print(msg, name)

greet(name="Alice", msg="Welcome")

Arbitrary Arguments


def add_all(*args):
    return sum(args)

print(add_all(1, 2, 3, 4))  # 10
Tip: Functions help organize code, avoid repetition, and make programs easier to read and maintain.
